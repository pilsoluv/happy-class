function updateTime() { const now = new Date(); const h = now.getHours(); const m = now.getMinutes(); const total = h * 60 + m; // 현재 시각을 분으로 환산 // 1. 상단 현재 시간 표시 (교시 이름 + 시간) const displayH = h % 12 || 12; const displayM = m < 10 ? '0' + m : m; // 교시 판별용 let periodName = ""; if (total < 520) periodName = "등교시간"; else if (total < 540) periodName = "아침활동"; else if (total < 580) periodName = "1교시"; else if (total < 590) periodName = "쉬는시간"; else if (total < 630) periodName = "2교시"; else if (total < 640) periodName = "쉬는시간"; else if (total < 680) periodName = "3교시"; else if (total < 740) periodName = "점심시간"; else if (total < 780) periodName = "4교시"; else if (total < 790) periodName = "쉬는시간"; else if (total < 830) periodName = "5교시"; else periodName = "일과종료"; // 상단 라인: [교시(노랑)] [시간(흰색)] document.getElementById("statusLine").innerHTML = `${periodName} ${displayH}시 ${displayM}분`; // 2. 하단 남은 시간 표시 let countdownText = ""; if (total < 830) { // [목표시간(분), 표시될 이름] - "시작" 글자 뺐습니다! const schedule = [ [520, "아침활동"], // 08:40 [540, "1교시"], // 09:00 [580, "쉬는시간"], // 09:40 [590, "2교시"], // 09:50 [630, "쉬는시간"], // 10:30 [640, "3교시"], // 10:40 [680, "점심시간"], // 11:20 [740, "4교시"], // 12:20 [780, "쉬는시간"], // 13:00 [790, "5교시"], // 13:10 [830, "하교"] // 13:50 ]; const target = schedule.find(item => total < item[0]); if (target) { const remain = target[0] - total; // 목표 이름과 남은 분수 모두 노란색(#ffca1b) 적용 countdownText = `${target[1]}까지 ${remain}분 남았습니다.`; } } else { countdownText = ""; // 1시 50분 이후엔 빈칸 } document.getElementById("countdownText").innerHTML = countdownText; }
